class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int,int> mp;
        for(int x:nums)
            mp[x]++;
        vector<pair<int,int>> v;
        for(auto x:mp)
            v.emplace_back(x);
        sort(v.begin(),v.end(),[](pair<int,int>& p1,pair<int,int>& p2){
            return p1.second>p2.second;
        });
        vector<int> res;
        for(int i=0;i<k;i++)
            res.emplace_back(v[i].first);
        return res;
    }
};
