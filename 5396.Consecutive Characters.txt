class Solution {
public:
    int maxPower(string s) {
        int res=1,i=1,cur=1;
        char x=s[0];
        while(i<s.size())
        {
            if(s[i]==x)
            {
                cur++;
                res=max(res,cur);
            }
            else
            {
                res=max(res,cur);
                x=s[i];
                cur=1;
            }
            i++;
        }
        return res;
    }
};
