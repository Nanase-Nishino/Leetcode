/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    void dfs(TreeNode* node)
    {
        if(!node||(!node->left&&!node->right))
            return;
        dfs(node->left);
        dfs(node->right);
        TreeNode* tem=node->left;
        node->left=node->right;
        node->right=tem;
    }
public:
    TreeNode* invertTree(TreeNode* root) {
        dfs(root);
        return root;
    }
};
