//每一位进行移动O(n*k)
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int N=nums.size();
        int temp,pre;
        for(int i=0;i<k;i++)
        {
            pre=nums[N-1];
            for(int j=0;j<N;j++)
            {
                temp=nums[j];
                nums[j]=pre;
                pre=temp;
            }
        }
    }
};
