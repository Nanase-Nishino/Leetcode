class Solution {
public:
    vector<int> smallerNumbersThanCurrent(vector<int>& nums) {

        int N=nums.size();
        vector<int> v;
        for(int i=0;i<N;i++)
        {
            int cnt=0;
            for(int j=0;j<N;j++)
            {
                if(nums[i]>nums[j])
                    cnt++;
            }
            v.push_back(cnt);
        }
        return v;
    }
};
