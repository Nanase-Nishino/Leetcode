#include<cstdio>
class Solution {
public:
    int gcd(int a,int b)
    {
        return b>0?gcd(b,a%b):a;
    }
    vector<string> simplifiedFractions(int n) {
        if(n<=1)
            return {};
        vector<string> res;
        for(int i=2;i<=n;i++)
            for(int j=1;j<i;j++)
            {
                if(gcd(i,j)!=1)
                    continue;
                else
                {
                    string s;
                    s+=(to_string(j));
                    s+='/';
                    s+=(to_string(i));
                    res.push_back(s);
                }
            }
        return res;
    }
};
